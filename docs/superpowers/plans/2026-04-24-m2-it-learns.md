// crates/neuron/src/lif.rs
use serde::{Deserialize, Serialize};

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct LifParams {
    pub threshold: f32,
    pub leak_rate: f32,
    pub reset_potential: f32,
}

impl Default for LifParams {
    fn default() -> Self {
        Self {
            threshold: 1.0,
            leak_rate: 0.05,
            reset_potential: 0.0,
        }
    }
}

/// Leaky Integrate-and-Fire step.
/// Adds input to potential, applies leak, checks threshold.
/// Returns true if the neuron fires (spikes).
pub fn integrate_and_fire(potential: &mut f32, input: f32, params: &LifParams) -> bool {
    // Leak
    *potential -= params.leak_rate;
    if *potential < 0.0 {
        *potential = 0.0;
    }

    // Integrate input
    *potential += input;

    // Fire check
    if *potential >= params.threshold {
        *potential = params.reset_potential;
        true
    } else {
        false
    }
}

// crates/neuron/src/stdp.rs
use serde::{Deserialize, Serialize};

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct StdpParams {
    pub window: u64,         // time window for plasticity
    pub a_plus: f32,         // max strengthening factor
    pub a_minus: f32,        // max weakening factor
    pub weight_max: f32,
    pub weight_min: f32,
}

impl Default for StdpParams {
    fn default() -> Self {
        Self {
            window: 20,
            a_plus: 0.02,
            a_minus: 0.01,
            weight_max: 1.0,
            weight_min: 0.0,
        }
    }
}

/// Compute STDP weight change based on spike timing.
/// pre_tick: when the presynaptic neuron last fired
/// post_tick: when the postsynaptic neuron last fired
/// Returns positive (strengthen) if pre fires before post, negative (weaken) if after.
pub fn stdp_weight_change(pre_tick: u64, post_tick: u64, params: &StdpParams) -> f32 {
    if pre_tick == post_tick {
        return 0.0;
    }

    let dt = post_tick as i64 - pre_tick as i64;
    let abs_dt = dt.unsigned_abs();

    if abs_dt > params.window {
        return 0.0;
    }

    let decay = 1.0 - (abs_dt as f32 / params.window as f32);

    if dt > 0 {
        // Pre before post -> strengthen
        params.a_plus * decay
    } else {
        // Post before pre -> weaken
        -params.a_minus * decay
    }
}

/// Apply STDP to a synapse weight, modulated by information gain.
/// Higher info_gain amplifies learning.
pub fn apply_stdp(
    weight: &mut f32,
    pre_tick: u64,
    post_tick: u64,
    info_gain: f32,
    params: &StdpParams,
) {
    let base_change = stdp_weight_change(pre_tick, post_tick, params);
    let modulated = base_change * (1.0 + info_gain);
    *weight = (*weight + modulated).clamp(params.weight_min, params.weight_max);
}

// crates/neuron/src/curiosity.rs

/// Compute information gain as the absolute prediction error, clamped to [0, 1].
/// This is how "surprised" a cell is by its input.
pub fn compute_information_gain(predicted: f32, actual: f32) -> f32 {
    (actual - predicted).abs().clamp(0.0, 1.0)
}

/// Update prediction toward actual value using an exponential moving average.
pub fn update_prediction(predicted: &mut f32, actual: f32, learning_rate: f32) {
    *predicted += learning_rate * (actual - *predicted);
}

/// Homeostatic curiosity drive.
/// Returns positive when info_gain is below target (seek novelty).
/// Returns negative when info_gain is above target (seek stability).
/// Returns 0 when at target.
pub fn homeostatic_curiosity_drive(current_info_gain: f32, target_info_gain: f32) -> f32 {
    target_info_gain - current_info_gain
}

/// Compute a sensory input value from local substrate state.
/// Converts resource_density into a [0, 1] signal that sensor cells receive.
pub fn substrate_to_signal(resource_density: f32) -> f32 {
    // Normalize: resource_density is roughly 0-5, map to 0-1
    (resource_density / 5.0).clamp(0.0, 1.0)
}

// crates/neuron/src/spike.rs
use organic_core::cell::{Cell, CellId};
use organic_core::direction::Direction;

/// Collect total synaptic input for a cell based on which presynaptic cells are spiking.
/// pre_spikes is a list of (cell_id, is_spiking) pairs from the previous tick.
pub fn collect_synaptic_input(cell: &Cell, pre_spikes: &[(CellId, bool)]) -> f32 {
    let mut total = 0.0;
    for synapse in &cell.connections {
        // Find if the source cell (synapse.target here means "connected to") is spiking
        for &(id, spiking) in pre_spikes {
            if id == synapse.target && spiking {
                total += synapse.weight;
            }
        }
    }
    total
}

/// Decode motor cell spike patterns into a movement direction.
/// Uses the last 4 spike states as a 4-bit direction encoding.
/// Returns None if no clear direction signal.
pub fn decode_motor_output(recent_spikes: &[bool]) -> Option<Direction> {
    if recent_spikes.is_empty() {
        return None;
    }

    // Simple scheme: count recent active spikes to pick a direction
    // Use the pattern hash to deterministically select a direction
    let pattern_value: u32 = recent_spikes
        .iter()
        .enumerate()
        .filter(|(_, &s)| s)
        .map(|(i, _)| 1u32 << (i % 4))
        .fold(0u32, |acc, v| acc | v);

    if pattern_value == 0 {
        return None;
    }

    let dirs = Direction::all();
    Some(dirs[(pattern_value as usize) % 4])
}

// crates/engine/src/neural_tick.rs
use organic_core::cell::{Cell, CellId, CellType};
use organic_core::direction::Direction;
use organic_core::genome::LearningParams;
use organic_neuron::curiosity::{compute_information_gain, substrate_to_signal, update_prediction};
use organic_neuron::lif::{integrate_and_fire, LifParams};
use organic_neuron::spike::{collect_synaptic_input, decode_motor_output};
use organic_neuron::stdp::{apply_stdp, StdpParams};
use organic_substrate::sal::{LocalState, SubstrateInterface};

use crate::energy;

/// Result of the neural tick for one organism.
pub struct NeuralTickResult {
    pub move_direction: Option<Direction>,
    pub energy_cost: f32,
    pub spike_count: u32,
}

/// Execute the 6-step neural cycle for all cells in an organism.
/// This replaces the hardcoded living behavior from M1.
pub fn run_neural_tick(
    cells: &mut [Cell],
    tick: u64,
    learning_params: &LearningParams,
    grid: &dyn SubstrateInterface,
) -> NeuralTickResult {
    let lif_params = LifParams::default();
    let stdp_params = StdpParams {
        window: learning_params.stdp_window as u64,
        ..StdpParams::default()
    };

    // Collect previous tick's spike state for all cells (1-tick delay)
    let prev_spikes: Vec<(CellId, bool)> = cells
        .iter()
        .map(|c| (c.id, c.spike_active))
        .collect();

    let mut spike_count = 0u32;
    let mut motor_spikes: Vec<bool> = Vec::new();
    let mut energy_cost = 0.0f32;

    for cell in cells.iter_mut() {
        // Step 1: SENSE
        let sensory_input = match cell.cell_type {
            CellType::Sensor => {
                let local: LocalState = grid.sense(cell.position);
                substrate_to_signal(local.resource_density)
            }
            _ => 0.0,
        };

        let synaptic_input = collect_synaptic_input(cell, &prev_spikes);
        let total_input = sensory_input + synaptic_input;

        // Step 2: PREDICT
        let info_gain = compute_information_gain(cell.predicted_input, total_input);
        cell.information_gain = info_gain;

        // Step 3: INTEGRATE + Step 4: FIRE
        let fired = integrate_and_fire(&mut cell.membrane_potential, total_input, &lif_params);
        cell.spike_active = fired;

        if fired {
            cell.last_spike_tick = Some(tick);
            spike_count += 1;
            energy_cost += energy::COST_SPIKE;
        }

        // Track motor spikes for movement decoding
        if cell.cell_type == CellType::Motor {
            motor_spikes.push(fired);
        }

        // Step 5: LEARN (STDP on all connections)
        if let Some(post_tick) = cell.last_spike_tick {
            for synapse in &mut cell.connections {
                // Find presynaptic spike time
                if let Some(pre_tick) = synapse.last_pre_tick {
                    apply_stdp(
                        &mut synapse.weight,
                        pre_tick,
                        post_tick,
                        info_gain * learning_params.learning_rate,
                        &stdp_params,
                    );
                }
            }
        }

        // Update presynaptic spike times on connections
        for synapse in &mut cell.connections {
            for &(id, spiking) in &prev_spikes {
                if id == synapse.target && spiking {
                    synapse.last_pre_tick = Some(tick);
                }
            }
        }

        // Decay unused connections
        for synapse in &mut cell.connections {
            if synapse.last_pre_tick.map_or(true, |t| tick - t > learning_params.stdp_window as u64 * 2) {
                synapse.weight *= 1.0 - learning_params.decay_rate;
            }
        }

        // Prune dead connections
        cell.connections.retain(|s| s.weight > 0.01);

        // Update prediction
        update_prediction(&mut cell.predicted_input, total_input, learning_params.learning_rate);
    }

    // Step 6: ACT — decode motor output
    let move_direction = if !motor_spikes.is_empty() {
        decode_motor_output(&motor_spikes)
    } else {
        None
    };

    NeuralTickResult {
        move_direction,
        energy_cost,
        spike_count,
    }
}

mod tests {
    use super::*;
    use organic_core::cell::{Cell, CellType, Synapse};

    fn make_organism_cells() -> Vec<Cell> {
        let mut sensor = Cell::new_seed(0, Position::new(5, 5));

        let mut inter = Cell::new_child(1, Position::new(5, 5), 1);
        inter.cell_type = CellType::Inter;
        inter.connections.push(Synapse { target: 0, weight: 0.5, last_pre_tick: None });

        let mut motor = Cell::new_child(2, Position::new(5, 5), 2);

    }

    fn test_neural_tick_runs_without_panic() {
        let mut cells = make_organism_cells();
        let params = LearningParams::default();
        let mut grid = Grid::new(20, 20);
        let mut rng = rand::thread_rng();
        grid.scatter_resources(50, 5.0, &mut rng);

        assert!(result.energy_cost >= 0.0);

    #[test]

        let mut cells = make_organism_cells();
        let params = LearningParams::default();
        let mut grid = Grid::new(10, 10);
        // Place resource at sensor position
        let mut rng = rand::thread_rng();
        grid.scatter_resources(100, 5.0, &mut rng); // Dense resources

        // Run several ticks to build up potential
        for t in 0..20 {
            run_neural_tick(&mut cells, t, &params, &grid);
        }

        // Sensor should have received input and potentially spiked
        // Check that prediction has moved from 0 (it learned something)
        assert!(cells[0].predicted_input != 0.0 || cells[0].information_gain >= 0.0);
    }

    #[test]
    fn test_stdp_modifies_weights_over_time() {
        let mut cells = make_organism_cells();
        let params = LearningParams { learning_rate: 0.1, ..LearningParams::default() };
        let mut grid = Grid::new(10, 10);
        let mut rng = rand::thread_rng();
        grid.scatter_resources(100, 5.0, &mut rng);

        let initial_weight = cells[1].connections[0].weight;

        for t in 0..50 {

                                cell.position = new_pos;
                            }
                            org.energy -= energy::COST_MOVE;
                        }
                    }

                    // Sensor cells consume resources
                    for cell in &org.cells {
                        if cell.cell_type == CellType::Sensor {

#[test]
fn test_learning_organisms_improve_over_lifetime() {
    let config = WorldConfig {
        width: 30,
        height: 30,
        initial_resource_count: 200,
        initial_organism_count: 5,
    };
    let mut world = World::new(config);

    // Force all organisms to Living phase with some cells
    for org in &mut world.organisms {
        org.phase = LifecyclePhase::Living;
        org.energy = 15.0;
    }

    // Track energy over time
    let mut early_energy: Vec<f32> = Vec::new();
    let mut late_energy: Vec<f32> = Vec::new();

    for tick in 0..2000 {
        world.tick();

        if world.organisms.is_empty() { break; }

        let avg_energy: f32 = world.organisms.iter().map(|o| o.energy).sum::<f32>()
            / world.organisms.len() as f32;

        if tick >= 100 && tick < 200 {
            early_energy.push(avg_energy);
        }
        if tick >= 1800 && tick < 2000 {
            late_energy.push(avg_energy);
        }
    }

    // We can't guarantee learning improves energy (stochastic), but the simulation
    // should run without panicking and organisms should survive
    println!(
        "Early avg energy: {:.2}, Late avg energy: {:.2}, Final organisms: {}",
        early_energy.iter().sum::<f32>() / early_energy.len().max(1) as f32,
        late_energy.iter().sum::<f32>() / late_energy.len().max(1) as f32,
        world.organisms.len()
    );

    // At minimum, the simulation ran 2000 ticks without panicking
    assert!(world.tick_count >= 2000 || world.organisms.is_empty());
}

#[test]
fn test_neural_activity_produces_spikes() {
    let config = WorldConfig {
        width: 20,
        height: 20,
        initial_resource_count: 100,
        initial_organism_count: 3,
    };
    let mut world = World::new(config);

    // Run until organisms are living
    for _ in 0..100 { world.tick(); }

    // Check that some cells have spiked
    let any_spikes = world.organisms.iter().any(|o| {
        o.cells.iter().any(|c| c.last_spike_tick.is_some())
    });

    let snap = world.snapshot();
    let any_info_gain = snap.organisms.iter().any(|o| {
        o.cells.iter().any(|c| c.information_gain > 0.0)
    });

    println!("Any spikes after 100 ticks: {}, Any info gain: {}", any_spikes, any_info_gain);
    // Neural activity should be happening
    // (may not if all organisms died, which is valid)
    assert!(any_spikes || world.organisms.is_empty(), "Living organisms should produce spikes");
}
